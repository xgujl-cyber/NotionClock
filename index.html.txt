    <script>
        function updateClock() {
            const now = new Date();
            const monthNames = ["JAN", "FEB", "MAR", "APR", "MAY", "JUN", "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"];
            const daysShort = ["SUN", "MON", "TUE", "WED", "THU", "FRI", "SAT"];
            const h = now.getHours(); const m = now.getMinutes();
            document.getElementById('display-year').innerText = now.getFullYear();
            document.getElementById('display-month-en').innerText = monthNames[now.getMonth()];
            document.getElementById('display-weekday-short').innerText = daysShort[now.getDay()];
            document.getElementById('display-m-d').innerText = (now.getMonth() + 1) + '/' + now.getDate();
            document.getElementById('display-time').innerText = String(h).padStart(2, '0') + ':' + String(m).padStart(2, '0');
            const startTime = 6; const totalMinutes = 18 * 60;
            let currentMinutes = (h >= startTime) ? (h - startTime) * 60 + m : 0;
            let progress = Math.min((currentMinutes / totalMinutes) * 100, 100);
            if (h < startTime) progress = 0;
            if (h >= 0 && h < 6) { progress = 100; } 
            document.getElementById('time-pointer').style.left = progress + '%';
            document.getElementById('progress-fill').style.width = progress + '%';
        }
        setInterval(updateClock, 1000);
        updateClock();
    </script>
